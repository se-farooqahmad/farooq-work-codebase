//iterative function to sum first n elements in an array
float sum(float a[],  int n)
    {
       float s =0;
       for (int i=0; i<n;i++)
           s=s+a[i];
        return s;
   }

//recursive function to sum first n elements in an array
float rsum(float a[],  int n)
{
       if(n<=0) return 0;
       else return (rsum(a, n-1) +a[n-1]);
}
